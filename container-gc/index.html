    FLAG_SET_ERGO_IF_DEFAULT(UseParallelGC, true);

    FLAG_SET_ERGO_IF_DEFAULT(UseSerialGC, true);

  } else {

    FLAG_SET_ERGO_IF_DEFAULT(UseSerialGC, true);

  }
